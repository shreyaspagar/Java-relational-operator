public class Relationaloperator {
    public static void main(String[] args) {
        int p = 45, q = 78;



      System.out.println(p == q); //False
      System.out.println(p != q); //True
      System.out.println(p > q);  //False
      System.out.println(p >= q); //False
      System.out.println(p < q);  //True
      System.out.println(p <= q); //True


    }

}
